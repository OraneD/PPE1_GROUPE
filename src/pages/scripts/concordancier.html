description du script

Script pour les corpus français, anglais et le corpus "štreber" ("nerd") serbe

#!/usr/bin/bash



fichier_text=$1

motif=$2


if [[ $# -ne 2 ]]

then

               echo "Ce programme demande exactement deux arguments."

               exit

fi


if [[ ! -f $fichier_text ]]

then

         echo "le fichier $fichier_text n'existe pas"

         exit

fi


if [[ -z $motif ]]

then

         echo "le motif est vide"

         exit

fi


echo "

<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <title>Concordance</title>

</head>

<body>

<table>

<thead>

    <tr>

        <th class="has-text-left">Contexte gauche</th>

        <<th>Cible</th>

        <<th class="has-text-right">Contexte droit</th>

    <</tr>

</thead>

<tbody>

"


grep -E -o "(\w+\W+){0,5}($motif)(\W+\w+){0,5}" $fichier_text | sed -E "s/(.*)($motif)(.*)/<tr><td>\1<\/td><td>\2<\/td><td>\4<\/td><\/tr>/"


echo "

</tbody>

</table>

</body>

</html>

"
